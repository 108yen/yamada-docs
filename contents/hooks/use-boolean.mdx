---
title: useBoolean
description: '`useBoolean` is a custom hook used to manage boolean values using `on`, `off`, and `toggle` functions.'
package_name: '@yamada-ui/use-boolean'
with_description: true
---

## Import

```ts
import { useBoolean } from '@yamada-ui/react'
```

## Usage

```tsx functional=true
const [flg, { on, off, toggle }] = useBoolean()

return (
  <>
    <Text>state: {String(flg)}</Text>

    <Wrap gap='md'>
      <Button mt='md' onClick={on}>
        On
      </Button>
      <Button mt='md' onClick={off}>
        Off
      </Button>
      <Button mt='md' onClick={toggle}>
        Toggle
      </Button>
    </Wrap>
  </>
)
```

### Use initial value

```tsx functional=true
const [flg, { on, off, toggle }] = useBoolean(true)

return (
  <>
    <Text>state: {String(flg)}</Text>

    <Wrap gap='md'>
      <Button mt='md' onClick={on}>
        On
      </Button>
      <Button mt='md' onClick={off}>
        Off
      </Button>
      <Button mt='md' onClick={toggle}>
        Toggle
      </Button>
    </Wrap>
  </>
)
```
