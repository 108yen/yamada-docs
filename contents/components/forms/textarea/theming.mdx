## Theming

The `Textarea` is a [single part component](/styled-system/theming/component-style#single-part-component).

:::note
`Textarea` inherits the style from [pickStyle(Input](undefined).
:::

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const Textarea: ComponentStyle = mergeStyle(pickStyle(Input, "field"), {
  baseStyle: {
    py: "2",
    lineHeight: "short",
    verticalAlign: "top",
  },

  variants: {
    unstyled: {
      h: "auto",
      py: "0",
      px: "0",
    },
  },

  sizes: {
    xs: {
      py: "2",
      minH: "20",
    },
    sm: {
      py: "2",
      minH: "20",
    },
    md: {
      py: "2",
      minH: "20",
    },
    lg: {
      py: "3",
      minH: "20",
    },
  },
})()
```
