```tsx live=false
// When to use theme token value.
<Box borderRadius='md' />

// When to use custom value.
<Box borderRadius='2rem' />

// When to use responsive styles.
<Box borderRadius={{ base: 'md', md: '8px' }} />

// When to use color mode styles.
<Box borderRadius={['md', 'sm']} />
```
